showUndoToast(removedItem) {
  // Remove any existing toast
  document.querySelector('.cart-undo-toast')?.remove();

  const toast = document.createElement('div');
  toast.className = 'cart-undo-toast';
  toast.innerHTML = `
    Item removed.
    <button type="button" class="cart-undo-btn">Undo</button>
  `;
  document.body.appendChild(toast);

  // Animate in
  requestAnimationFrame(() => toast.classList.add('is-visible'));

  const undoBtn = toast.querySelector('.cart-undo-btn');
  let timeout = setTimeout(() => {
    toast.classList.remove('is-visible');
    setTimeout(() => toast.remove(), 300);
  }, 5000);

  undoBtn.addEventListener('click', () => {
    clearTimeout(timeout);
    // Re-add the item via the Cart API
    fetch('/cart/add.js', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        id: removedItem.variantId,
        quantity: parseInt(removedItem.quantity)
      })
    })
    .then(() => location.reload())
    .catch(() => toast.remove());
    toast.remove();
  });
}

assets/cart.js — add as a global function

function swapCartVariant(select) {
  const line = select.dataset.line;
  const newVariantId = select.value;
  const currentVariantId = select.dataset.currentVariant;
  if (newVariantId === currentVariantId) return;

  // Get current quantity
  const qtyInput = document.querySelector(
    `#Quantity-${line}, #Drawer-quantity-${line}`
  );
  const quantity = qtyInput ? parseInt(qtyInput.value) : 1;

  // Remove old variant, then add new one
  fetch('/cart/change.js', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ line: parseInt(line), quantity: 0 })
  })
  .then(() => fetch('/cart/add.js', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ id: parseInt(newVariantId), quantity })
  }))
  .then(() => location.reload());
}

1 Swipe-to-remove on cart itemsMedium Impact

Add horizontal swipe detection on cart items in the drawer. When swiped left, reveal a red "Remove" background and remove the item.

assets/cart.js — add at the end of the file

// Swipe-to-remove for mobile cart items
if ('ontouchstart' in window) {
  document.addEventListener('touchstart', function(e) {
    const item = e.target.closest('.cart-item');
    if (!item || !item.closest('cart-drawer')) return;

    const startX = e.touches[0].clientX;
    const startY = e.touches[0].clientY;
    let moved = false;

    function onMove(e) {
      const dx = e.touches[0].clientX - startX;
      const dy = e.touches[0].clientY - startY;
      if (Math.abs(dy) > Math.abs(dx)) return; // vertical scroll
      if (dx < -30) {
        moved = true;
        item.style.transform = `translateX(${Math.max(dx, -100)}px)`;
        item.style.transition = 'none';
      }
    }

    function onEnd() {
      document.removeEventListener('touchmove', onMove);
      document.removeEventListener('touchend', onEnd);
      if (moved) {
        const currentX = parseInt(item.style.transform.replace(/[^-\d]/g, '')) || 0;
        if (currentX < -60) {
          item.style.transition = 'transform 0.2s, opacity 0.2s';
          item.style.transform = 'translateX(-100%)';
          item.style.opacity = '0';
          setTimeout(() => {
            const removeBtn = item.querySelector('cart-remove-button button');
            if (removeBtn) removeBtn.click();
          }, 200);
        } else {
          item.style.transition = 'transform 0.2s';
          item.style.transform = '';
        }
      }
    }

    document.addEventListener('touchmove', onMove, { passive: true });
    document.addEventListener('touchend', onEnd);
  }, { passive: true });
}

// Discount code AJAX validation
document.addEventListener('click', function(e) {
  const applyBtn = e.target.closest('[id$="DiscountApply"]');
  if (!applyBtn) return;

  const container = applyBtn.closest('[class*="discount"]');
  const input = container.querySelector('input[type="text"]');
  const feedback = container.querySelector('[id$="DiscountFeedback"]');
  const code = input?.value?.trim();

  if (!code) {
    feedback.innerHTML = '<span style="color:#c62828">Please enter a code</span>';
    return;
  }

  applyBtn.disabled = true;
  applyBtn.textContent = 'Applying...';

  // Step 1: Apply discount via Shopify's discount URL (sets cookie)
  fetch(`/discount/${encodeURIComponent(code)}`, { redirect: 'follow' })
    .then(() => fetch('/cart.js'))
    .then(r => r.json())
    .then(cart => {
      // Check if any cart-level discount matches our code
      const applied = cart.cart_level_discount_applications?.some(
        d => d.title.toLowerCase() === code.toLowerCase()
      );
      if (applied || cart.total_discount > 0) {
        feedback.innerHTML =
          '<span style="color:#2e7d32;font-weight:600">' +
          '✓ Code applied! Saving ' + (cart.total_discount / 100).toFixed(2) +
          '</span>';
        // Reload to show updated prices
        setTimeout(() => location.reload(), 1000);
      } else {
        feedback.innerHTML =
          '<span style="color:#c62828">This code is not valid or has expired</span>';
      }
      applyBtn.disabled = false;
      applyBtn.textContent = 'Apply';
    })
    .catch(() => {
      feedback.innerHTML =
        '<span style="color:#c62828">Something went wrong — please try again</span>';
      applyBtn.disabled = false;
      applyBtn.textContent = 'Apply';
    });
});

function quickAddToCart(btn) {
  const variantId = btn.dataset.variantId;
  btn.disabled = true;
  btn.textContent = 'Adding...';

  fetch('/cart/add.js', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ items: [{ id: parseInt(variantId), quantity: 1 }] })
  })
  .then(r => {
    if (!r.ok) throw new Error('Failed');
    return r.json();
  })
  .then(() => {

// Track recently viewed products (call this on PDP pages)
function trackRecentlyViewed(productData) {
  const MAX = 10;
  let recent = JSON.parse(localStorage.getItem('recently_viewed') || '[]');
  // Remove if already exists, then add to front
  recent = recent.filter(p => p.id !== productData.id);
  recent.unshift(productData);
  recent = recent.slice(0, MAX);
  localStorage.setItem('recently_viewed', JSON.stringify(recent));
}

// Display recently viewed in empty cart state
function renderRecentlyViewed(container) {
  const recent = JSON.parse(localStorage.getItem('recently_viewed') || '[]');
  if (!recent.length) return;

  const html = `
    <div class="empty-cart__recently-viewed">
      <h3 class="empty-cart__section-heading">Recently Viewed</h3>
      <div class="empty-cart__products-grid">
        ${recent.slice(0, 4).map(p => `
          <a href="${p.url}" class="empty-cart__product">
            <img src="${p.image}" alt="${p.title}" width="120" height="120"
                 loading="lazy" style="object-fit:cover;border-radius:6px;">
            <span class="empty-cart__product-title">${p.title.split(' | ')[0]}</span>
            <span class="empty-cart__product-price">${p.price}</span>
          </a>
        `).join('')}
      </div>
    </div>
  `;
  container.insertAdjacentHTML('beforeend', html);
}

assets/cart.js — countdown logic

// Sale countdown timer
document.querySelectorAll('.cart-sale-countdown').forEach(el => {
  const endDate = new Date(el.dataset.end).getTime();
  const timerEl = el.querySelector('[id$="CountdownTimer"]');

  function update() {
    const now = Date.now();
    const diff = endDate - now;
    if (diff <= 0) {
      el.remove();
      return;
    }
    const h = Math.floor(diff / 3600000);
    const m = Math.floor((diff % 3600000) / 60000);
    const s = Math.floor((diff % 60000) / 1000);
    timerEl.textContent =
      `${h}h ${String(m).padStart(2,'0')}m ${String(s).padStart(2,'0')}s`;
  }

  update();
  setInterval(update, 1000);
});
